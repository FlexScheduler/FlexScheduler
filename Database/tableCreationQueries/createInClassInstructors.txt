CREATE TABLE IF NOT EXISTS InClassInstructors(
    deptID VARCHAR(16),
    crseID INT,
    secID VARCHAR(16),
    dayID CHAR(1),
    timeFrm TIME,
    timeTo TIME,
    instrFName VARCHAR(32),
    instrLName VARCHAR(32),
    PRIMARY KEY(
        deptID,
        crseID,
        secID,
        dayID,
        timeFrm,
        timeTo,
        instrFName,
        instrLName
    ),
    FOREIGN KEY(
        deptID,
        crseID,
        secID,
        dayID,
        timeFrm,
        timeTo
    ) REFERENCES InClassSessions(
        deptID,
        crseID,
        secID,
        dayID,
        timeFrm,
        timeTo
    )
);
