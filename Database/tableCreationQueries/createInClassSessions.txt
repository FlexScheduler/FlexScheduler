CREATE TABLE IF NOT EXISTS InClassSessions(
    deptID VARCHAR(16),
    crseID INT,
    secID VARCHAR(16),
    dayID CHAR(1),
    timeFrm TIME,
    timeTo TIME,
    rmAbbrv VARCHAR(16) NOT NULL,
    rmNum VARCHAR(32) NOT NULL,
    PRIMARY KEY(
        deptID,
        crseID,
        secID,
        dayID,
        timeFrm,
        timeTo
    ),
    FOREIGN KEY(deptID, crseID, secID) REFERENCES Sections(deptID, crseID, secID)
);
