CREATE TABLE IF NOT EXISTS Sections(
    deptID VARCHAR(16),
    crseID INT,
    secID VARCHAR(16),
    strtDate DATE NOT NULL,
    endDate DATE NOT NULL,
    campus VARCHAR(32) NOT NULL,
    PRIMARY KEY(deptID, crseID, secID),
    FOREIGN KEY(deptID, crseID) REFERENCES Courses(deptID, crseID)
);
